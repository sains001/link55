async function fetchData(){ try{ const res = await fetch("/api?t=" + Date.now()); const data = await res.json(); if(!data || !data.periode) return; if(data.periode !== lastPeriode){ // ========================= // HITUNG STATUS PERIODE LAMA // ========================= if(lastPeriode !== null && historyData.length > 0){ const hasilKategori = kategori(data.hasil); const status = (currentPrediction === hasilKategori) ? "WIN" : "LOSE"; historyData[0].hasil = data.hasil; historyData[0].status = status; } // ========================= // BUAT PREDIKSI BARU // ========================= currentPrediction = randomPrediksi(); countNumber++; historyData.unshift({ periode:data.periode, prediksi:currentPrediction, count:countNumber, hasil:"-", status:"-" }); if(historyData.length > 5){ historyData.pop(); } lastPeriode = data.periode; document.getElementById("periode").innerText = data.periode; document.getElementById("prediksi").innerText = currentPrediction; renderTable(); } }catch(e){ console.log("Fetch error:", e); } }
